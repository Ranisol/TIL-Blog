# Powerset

- Question
주어진 문자열의 'power set'으로 이루어진 배열을 return하는 함수를 작성하세요.

- solution

/ for문을 재귀적으로 표현한다면?

const powerSet = function (str, powerSetArr) {
  let arr = str.split('')
  arr = Array.from(new Set(arr))
  let powersetArr = []
  let subset = function(start, num, args){
    for(let a = start; a < arr.length; a++){
      args = args || ''
      powersetArr.push(args + arr[a])
      if(num > 0){
        let newargs = args + arr[a]
        num --;
        subset(a+1,num, newargs)
      }
    }
  }
  subset(0,arr.length)
  powersetArr.push('')
  return powersetArr
};
